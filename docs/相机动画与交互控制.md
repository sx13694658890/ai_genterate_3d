一、 相机运动动画的艺术与实现
1.1 直线运动:模拟飞鸟视角
直线运动是相机动画中最基础也最实用的形式,适用于展现广阔场景或模拟高空俯瞰效果。核心原理是通过持续更新相机的position属性来创造动态视角。
图片当相机在空中直线飞行时,能够完美呈现工厂的全景布局。下面是实现直线运动的核心代码:
// 创建透视投影相机
const camera = new THREE.PerspectiveCamera(
    60,  // 视场角
    window.innerWidth / window.innerHeight,  // 宽高比
    0.1,  // 近剪裁面
    2000  // 远剪裁面
);
// 设置初始位置 - 位于高空俯视
camera.position.set(0, 100, 200);
// 渲染循环中实现相机直线运动
let progress = 0;
const speed = 0.5;  // 运动速度
function animate() {
    // 相机沿Z轴负方向持续运动
    camera.position.z -= speed;
    // 确保相机始终朝向观察目标
    camera.lookAt(0, 0, 0);
    renderer.render(scene, camera);
    requestAnimationFrame(animate);
}

1.2 圆周运动:全方位展示建筑
当需要全方位展示一个建筑或模型时,圆周运动是最佳选择。通过三角函数计算相机在XOZ平面上的位置,可以实现平滑的环绕效果。
// 圆周运动参数配置
const orbitRadius = 150;  // 圆周半径
const orbitSpeed = 0.005;  // 运动速度
let currentAngle = 0;  // 当前角度
// 圆周运动渲染函数
function circularOrbit() {
    // 更新角度
    currentAngle += orbitSpeed;
    // 计算相机新位置
    camera.position.x = orbitRadius * Math.cos(currentAngle);
    camera.position.z = orbitRadius * Math.sin(currentAngle);
    camera.position.y = 80;  // 保持一定高度
    // 相机始终朝向中心点
    camera.lookAt(0, 0, 0);
    renderer.render(scene, camera);
    requestAnimationFrame(circularOrbit);
}

1.3 lookAt()的精妙应用
在相机运动过程中,lookAt()方法起到了至关重要的作用。它不仅指定相机的观察方向,还决定了用户的视觉焦点。
关键要点:
lookAt()
必须在position更新后调用才能生效
如果不调用lookAt(),相机会保持默认方向不变
可以动态改变观察点来实现更复杂的效果
// 动态观察点示例
const observationPoints = [
    new THREE.Vector3(0, 0, 0),
    new THREE.Vector3(50, 20, 30),
    new THREE.Vector3(-40, 30, -20)
];
let targetIndex = 0;
function dynamicLookAt() {
    // 每300帧切换一次观察点
    if (renderer.info.render.frame % 300 === 0) {
        targetIndex = (targetIndex + 1) % observationPoints.length;
    }
    camera.lookAt(observationPoints[targetIndex]);
    renderer.render(scene, camera);
    requestAnimationFrame(dynamicLookAt);
}
二、 相机姿态控制:.up属性的妙用
2.1 理解.up的物理意义
.up属性定义了相机的"上方向",这决定了整个画面的旋转角度。默认情况下,相机的上方向指向Y轴正方向(0, 1, 0),这符合我们对世界的基本认知。
改变.up属性,就像你在拍照时旋转相机身体一样,会直接影响渲染结果的旋转状态。
2.2 不同up值的视觉效果
// 默认状态 - Y轴朝上
camera.up.set(0, 1, 0);
// 渲染效果:正常的3D视角,Y轴垂直向上
// Y轴朝下 - 整个画面倒置
camera.up.set(0, -1, 0);
camera.lookAt(0, 0, 0);  // 需要重新计算
// 渲染效果:画面上下颠倒
// X轴朝上 - 画面旋转90度
camera.up.set(1, 0, 0);
camera.lookAt(0, 0, 0);
// 渲染效果:红色X轴向上,整体画面顺时针旋转
// Z轴朝上 - 另一种旋转视角
camera.up.set(0, 0, 1);
camera.lookAt(0, 0, 0);
// 渲染效果:蓝色Z轴向上,画面呈现独特倾斜效果
2.3 执行顺序的重要性
一个常见的陷阱是:先执行lookAt(),再修改.up属性,然后忘记重新调用lookAt()。这会导致相机姿态计算错误。
// 正确的执行顺序
camera.up.set(1, 0, 0);      // 先设置上方向
camera.lookAt(0, 0, 0);      // 再计算相机姿态
// 或者
camera.lookAt(0, 0, 0);      // 先设置目标点
camera.up.set(1, 0, 0);      // 改变上方向
camera.lookAt(0, 0, 0);      // 重新计算姿态(关键!)

// 1. 定义管道路径 - 一系列三维控制点
const pathPoints = [
    new THREE.Vector3(-80, 50, 100),
    new THREE.Vector3(-30, 80, 50),
    new THREE.Vector3(0, 0, 0),
    new THREE.Vector3(80, -80, 0),
    new THREE.Vector3(120, -60, 80),
    new THREE.Vector3(150, 50, 40),
    new THREE.Vector3(200, 100, 0)

// 2. 创建样条曲线
const tubePath = new THREE.CatmullRomCurve3(pathPoints, {
    closed: false,        // 不闭合

// 3. 生成管道几何体

    12,           // 径向分段数

3.3 相机沿轨迹运动的核心逻辑
实现管道漫游的关键是让相机沿着管道路径移动,并且相机的观察方向要与路径切线方向一致。
// 从曲线上获取均匀分布的路径点
const pathPoints = tubePath.getSpacedPoints(500);
// 相机运动参数
let currentPointIndex = 0;
const movementSpeed = 2;  // 运动速度
// 管道漫游渲染函数
function pipeWanderAnimation() {
    // 确保不超过路径点范围
    if (currentPointIndex < pathPoints.length - 2) {
        // 获取当前点和下一个点
        const currentPosition = pathPoints[currentPointIndex];
        const nextPosition = pathPoints[currentPointIndex + 1];
        // 设置相机位置
        camera.position.copy(currentPosition);
        // 让相机看向下一个点,实现沿切线方向前进
        camera.lookAt(nextPosition);
        // 更新索引点
        currentPointIndex += movementSpeed;
    } else {
        // 到达终点后重新开始
        currentPointIndex = 0;
    }
    renderer.render(scene, camera);
    requestAnimationFrame(pipeWanderAnimation);
}
// 启动动画

五、 MapControls:地图交互的完美方案
MapControls是专门为地图类3D应用设计的控制器,提供了更符合地图交互习惯的操作方式。
5.1 MapControls的操作方式
MapControls与OrbitControls的最大区别在于鼠标操作的重新定义:
平移: 鼠标左键拖动(与OrbitControls的旋转相反)
旋转: 鼠标右键拖动(与OrbitControls的平移相反)
缩放: 鼠标中键滚动(与OrbitControls相同)
这种操作方式更符合Google Maps、百度地图等传统地图应用的交互习惯。

// 地图导航示例
class MapNavigator {
    constructor(camera, renderer, scene) {
        this.camera = camera;
        this.renderer = renderer;
        this.scene = scene;
        this.initControls();
        this.setupEventListeners();
    }
    initControls() {
        // 初始化MapControls
        this.controls = new MapControls(this.camera, this.renderer.domElement);
        // 配置适合地图的参数
        this.controls.enableDamping = true;
        this.controls.dampingFactor = 0.05;
        this.controls.maxPolarAngle = Math.PI / 2 - 0.05; // 限制垂直旋转
        this.controls.minDistance = 50;
        this.controls.maxDistance = 500;
    }
    setupEventListeners() {
        // 监听控制变化
        this.controls.addEventListener('change', () => {
            this.onControlsChange();
        });
        // 地点跳转功能
        this.setupLocationMarkers();
    }
    onControlsChange() {
        // 计算相机位置
        const position = this.camera.position;
        const target = this.controls.target;
        // 更新UI显示
        this.updateLocationDisplay(position, target);
        // 触发位置变化事件
        this.dispatchEvent({
            type: 'locationChange',
            position: position,
            target: target
        });
    }
    jumpToLocation(x, z, height = 150) {
        // 动画过渡到新位置
        const startPosition = this.camera.position.clone();
        const endPosition = new THREE.Vector3(x, height, z);
        const startTarget = this.controls.target.clone();
        const endTarget = new THREE.Vector3(x, 0, z);
        // 使用补间动画实现平滑过渡
        const duration = 1000; // 1秒动画
        const startTime = Date.now();
        function animateJump() {
            const elapsed = Date.now() - startTime;
            const progress = Math.min(elapsed / duration, 1);
            const easedProgress = this.easeInOutCubic(progress);
            // 线性插值计算当前位置
            this.camera.position.lerpVectors(startPosition, endPosition, easedProgress);
            this.controls.target.lerpVectors(startTarget, endTarget, easedProgress);
            this.controls.update();
            if (progress < 1) {
                requestAnimationFrame(animateJump.bind(this));
            }
        }
        animateJump.bind(this)();
    }
    easeInOutCubic(t) {
        return t < 0.5 ? 4 * t * t * t : 1 - Math.pow(-2 * t + 2, 3) / 2;
    }
    setupLocationMarkers() {
        // 在地图上设置位置标记点
        const locations = [
            { name: '工厂A区', x: -100, z: 50 },
            { name: '工厂B区', x: 100, z: -50 },
            { name: '设备中心', x: 0, z: 0 }
        ];
        locations.forEach(loc => {
            this.createLocationMarker(loc);
        });
    }
    createLocationMarker(location) {
        // 创建位置标记
        const markerGeometry = new THREE.ConeGeometry(5, 20, 8);
        const markerMaterial = new THREE.MeshStandardMaterial({
            color: 0xff0000,
            emissive: 0x330000
        });
        const marker = new THREE.Mesh(markerGeometry, markerMaterial);
        marker.position.set(location.x, 10, location.z);
        marker.userData = location; // 存储位置信息
        this.scene.add(marker);
        // 点击事件处理
        marker.addEventListener('click', () => {
            this.jumpToLocation(location.x, location.z);
        });
    }
    updateLocationDisplay(position, target) {
        // 更新位置显示UI
        const display = document.getElementById('location-display');
        if (display) {
            const lat = (target.z / 1000 * 90).toFixed(2);  // 模拟纬度
            const lon = (target.x / 1000 * 180).toFixed(2); // 模拟经度
            const altitude = position.y.toFixed(0);         // 高度
            display.textContent = `位置: ${lat}°, ${lon}° | 高度: ${altitude}m`;
        }
    }
}
// 使用示例

            color: color,
            roughness: 0.7,
            metalness: 0.3

        building.position.copy(position);
        building.castShadow = true;
        building.receiveShadow = true;

    addWindows(building, size) {
        const windowGeometry = new THREE.PlaneGeometry(3, 4);
        const windowMaterial = new THREE.MeshStandardMaterial({
            color: 0xffffaa,
            emissive: 0x443300,
            emissiveIntensity: 0.8
        });
        // 在建筑表面添加窗户
        for (let x = -size.x/2 + 10; x < size.x/2; x += 15) {
            for (let y = 5; y < size.y - 5; y += 12) {
                // 正面窗户
                const windowFront = new THREE.Mesh(windowGeometry, windowMaterial);
                windowFront.position.set(x, y - size.y/2, size.z/2 + 0.1);
                building.add(windowFront);
                // 背面窗户
                const windowBack = windowFront.clone();
                windowBack.position.z = -size.z/2 - 0.1;
                windowBack.rotation.y = Math.PI;
                building.add(windowBack);
            }
        }

            this.controls.dispose();

                this.setupOrbitMode();

                break;

    setupOrbitMode() {
        this.controls = new OrbitControls(this.camera, this.renderer.domElement);
        this.controls.enableDamping = true;
        this.controls.maxPolarAngle = Math.PI / 2 - 0.1;
        this.camera.position.set(0, 150, 300);
        this.controls.target.set(0, 0, 0);
        this.controls.update();
        this.flyoverProgress = 0;
        this.pipeIndex = 0;
    }
    setupFlyoverMode() {
        this.controls = null;
        this.camera.position.set(0, 200, -300);
        this.camera.lookAt(0, 0, 0);
        this.flyoverProgress = 0;
        this.pipeIndex = 0;
    }
    setupPipeMode() {
        this.controls = new OrbitControls(this.camera, this.renderer.domElement);
        this.controls.enableRotate = false;
        this.controls.enablePan = false;
        this.controls.enableZoom = false;
        this.pipeIndex = 0;
        this.flyoverProgress = 0;
        // 设置初始位置
        if (this.pathPoints && this.pathPoints.length > 0) {
            this.camera.position.copy(this.pathPoints[0]);
            this.camera.lookAt(this.pathPoints[1]);
        }
    }
    setupMapMode() {
        this.controls = new MapControls(this.camera, this.renderer.domElement);
        this.controls.enableDamping = true;
        this.controls.maxPolarAngle = Math.PI / 2 - 0.05;
        this.controls.minDistance = 50;
        this.controls.maxDistance = 400;
        this.camera.position.set(0, 300, 0);
        this.camera.lookAt(0, 0, 0);
        this.flyoverProgress = 0;
        this.pipeIndex = 0;
    }
    startAnimation() {
        this.animate();
    }
    animate() {
        requestAnimationFrame(() => this.animate());
        switch(this.animationMode) {
            case 'flyover':
                this.updateFlyoverAnimation();
                break;
            case 'pipe':
                this.updatePipeAnimation();
                break;
        }
        if (this.controls) {
            this.controls.update();
        }
        this.renderer.render(this.scene, this.camera);
    }
    updateFlyoverAnimation() {
        this.flyoverProgress += 0.002;
        // 相机直线飞行
        this.camera.position.z = -300 + this.flyoverProgress * 500;
        this.camera.position.x = Math.sin(this.flyoverProgress * 3) * 50;
        this.camera.lookAt(0, 0, 0);
        // 循环动画
        if (this.flyoverProgress > 1) {
            this.flyoverProgress = 0;
        }
    }
    updatePipeAnimation() {
        if (this.pathPoints && this.pipeIndex < this.pathPoints.length - 2) {
            const speed = 2;
            const currentPos = this.pathPoints[this.pipeIndex];
            const nextPos = this.pathPoints[this.pipeIndex + 1];
            this.camera.position.copy(currentPos);
            this.camera.lookAt(nextPos);
            this.pipeIndex += speed;
            // 同步控制器目标
            if (this.controls) {
                this.controls.target.copy(nextPos);
                this.controls.update();
            }
        } else {
            // 重置到起点
            this.pipeIndex = 0;
        }
    }
    resize() {
        const width = this.container.clientWidth;
        const height = this.container.clientHeight;
        this.camera.aspect = width / height;
        this.camera.updateProjectionMatrix();
        this.renderer.setSize(width, height);
    }

// 1. 使用时钟控制动画速度
const clock = new THREE.Clock();
const animationSpeed = 1.0;
function animate() {
    const delta = clock.getDelta(); // 获取两帧之间的时间间隔
    // 根据时间间隔计算移动距离,保证不同帧率下速度一致
    camera.position.z -= delta * animationSpeed * 50;
    requestAnimationFrame(animate);
}
// 2. 使用对象池减少垃圾回收
const vectorPool = [];
function getVector(x, y, z) {
    if (vectorPool.length > 0) {
        const vec = vectorPool.pop();
        vec.set(x, y, z);
        return vec;
    }
    return new THREE.Vector3(x, y, z);
}
function releaseVector(vec) {
    vectorPool.push(vec);
}
// 3. 减少lookAt()调用频率
let frameCounter = 0;
function optimizedAnimate() {
    frameCounter++;
    camera.position.z -= 0.3;
    // 每3帧才调用一次lookAt,减少计算量
    if (frameCounter % 3 === 0) {
        camera.lookAt(0, 0, 0);
    }
    requestAnimationFrame(optimizedAnimate);
}

function cleanup() {
    // 清理几何体
    geometry.dispose();
    // 清理材质
    material.dispose();
    // 清理纹理
    if (material.map) {
        material.map.dispose();
    }
    // 清理控制器
    if (controls) {
        controls.dispose();
    }
    // 清理渲染器

    }

// 场景切换时的资源管理
class SceneManager {
    constructor() {
        this.currentScene = null;
        this.loadedScenes = new Map();
    }
    loadScene(sceneName, loadCallback) {
        if (this.loadedScenes.has(sceneName)) {
            // 场景已存在,直接使用
            this.switchScene(this.loadedScenes.get(sceneName));
            return;
        }
        // 加载新场景
        loadCallback((scene) => {
            this.loadedScenes.set(sceneName, scene);
            this.switchScene(scene);
        });
    }
    switchScene(newScene) {
        if (this.currentScene) {
            // 清理当前场景
            this.currentScene.traverse((object) => {
                if (object.isMesh) {
                    object.geometry.dispose();
                    if (Array.isArray(object.material)) {
                        object.material.forEach(material => material.dispose());
                    } else {
                        object.material.dispose();
                    }
                }
            });
        }
        this.currentScene = newScene;
    }
    dispose() {
        // 清理所有缓存场景
        this.loadedScenes.forEach((scene) => {
            scene.traverse((object) => {
                if (object.isMesh) {
                    object.geometry.dispose();
                    if (Array.isArray(object.material)) {
                        object.material.forEach(material => material.dispose());
                    } else {
                        object.material.dispose();
                    }
                }
            });
        });
        this.loadedScenes.clear();
    }
}

        this.container = container;
        this.aspectRatio = container.clientWidth / container.clientHeight;
        this.pixelRatio = Math.min(window.devicePixelRatio, 2);
        this.setupCamera();
        this.setupRenderer();
        this.setupEventListeners();
    }
    setupCamera() {
        this.camera = new THREE.PerspectiveCamera(
            60,
            this.aspectRatio,
            0.1,
            2000
        );
    }
    setupRenderer() {
        this.renderer = new THREE.WebGLRenderer({
            antialias: true,
            alpha: true
        });
        this.renderer.setSize(
            this.container.clientWidth,
            this.container.clientHeight
        );
        this.renderer.setPixelRatio(this.pixelRatio);
        this.renderer.outputColorSpace = THREE.SRGBColorSpace;
        this.container.appendChild(this.renderer.domElement);
    }
    setupEventListeners() {
        // 窗口大小变化
        window.addEventListener('resize', () => this.onResize());
        // 容器大小变化(使用ResizeObserver)
        const resizeObserver = new ResizeObserver(() => this.onResize());
        resizeObserver.observe(this.container);
    }
    onResize() {
        const width = this.container.clientWidth;
        const height = this.container.clientHeight;
        // 更新宽高比
        this.aspectRatio = width / height;
        // 更新相机
        this.camera.aspect = this.aspectRatio;
        this.camera.updateProjectionMatrix();
        // 更新渲染器
        this.renderer.setSize(width, height);
        // 可选:根据屏幕大小调整渲染质量
        this.adjustQuality();
    }
    adjustQuality() {
        const width = this.container.clientWidth;
        // 移动设备降低像素比以提升性能
        if (width < 768) {
            this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 1.5));
        } else {
            this.renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        }
        // 根据设备性能调整阴影质量
        if (width < 768) {
            this.renderer.shadowMap.type = THREE.BasicShadowMap;
        } else {
            this.renderer.shadowMap.type = THREE.PCFSoftShadowMap;
        }
    }
}
